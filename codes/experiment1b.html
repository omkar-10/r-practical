# Experiment 1b – Execute Hadoop HDFS Commands

## Check Hadoop version
hadoop version

## List root directory contents
hdfs dfs -ls /

## Create a new directory
hdfs dfs -mkdir /input

## Copy file from local to HDFS
hdfs dfs -copyFromLocal /home/cloudera/Desktop/sample.txt /input

## OR (alternative)
hdfs dfs -put /home/cloudera/Desktop/sample.txt /input

## Display file contents
hdfs dfs -cat /input/sample.txt

## Copy file back from HDFS to local
hdfs dfs -copyToLocal /input/sample.txt /home/cloudera/Desktop/

## OR
hdfs dfs -get /input/sample.txt /home/cloudera/Desktop/

## Move file to another directory
hdfs dfs -mv /input/sample.txt /output/

## Copy file to another directory
hdfs dfs -cp /output/sample.txt /backup/

## Remove file
hdfs dfs -rm /backup/sample.txt

## Remove directory
hdfs dfs -rmdir /backup

## TIPS:
# Use hdfs dfs -help for quick command descriptions
