# Experiment 2 – Use Sqoop to Transfer Data Between Hadoop and MySQL

## 1. Verify MySQL Connection:
mysql -u root -p
# then type password (default: root)
show databases;
use testdb;
show tables;

## 2. Import Table from MySQL → HDFS:
sqoop import \
--connect jdbc:mysql://localhost/testdb \
--username root --password root \
--table employees \
--target-dir /user/hadoop/employees_data \
--m 1

# Explanation:
# --m 1 → single mapper for simple import
# --target-dir → where to save in HDFS

## 3. Import with Condition:
sqoop import \
--connect jdbc:mysql://localhost/testdb \
--username root --password root \
--table employees \
--where "salary > 50000" \
--target-dir /user/hadoop/highsalary

## 4. Export HDFS Data → MySQL:
sqoop export \
--connect jdbc:mysql://localhost/testdb \
--username root --password root \
--table employees_backup \
--export-dir /user/hadoop/employees_data \
--input-fields-terminated-by ','

## TIPS:
# Check imported data:
hdfs dfs -ls /user/hadoop/employees_data
